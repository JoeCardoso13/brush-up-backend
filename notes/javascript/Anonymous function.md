Any [[Function]] whose [[Function definition]] doesn't give it a name:
```js
function (args) { // function declaration
  body;
}

let myVar = function (args) { // function expression
  body;
}

const arrow = (args) => { // arrow function
  body;  
}
```
[[Arrow function]]s are **always** anonymous.
